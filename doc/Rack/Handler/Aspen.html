def self.run( app, options={} )
  # TODO: this sucks, server doesn't handle Rack::URLMap
  app = Rack::Chunked.new( Rack::ContentLength.new( app ) )
  server = ::Aspen::Server.new( options[:Host] || '0.0.0.0',
                                options[:Port] || 1169,
                                app )
  yield server if block_given?
  server.start
end
